25 {
26 
27  AWPDOUBLE max, min;
28 
29  _AWP_NORM_RGB_
30 
31  max = r > g ? r : g;
32  max = max > b ? max : b;
33  min = r < g ? r : g;
34  min = min < b ? min : b;
35 
36  *v = max;
37  if (max == 0)
38  {
39  *s = 0;
40  }
41  else
42  {
43  *s = 1 - min / max;
44  }
45 
46  if (max == min)
47  {
48  *h = 0;
49  }
50  else if (max == r && g >= b)
51  {
52  *h = 60 * (g - b) / (max - min);
53  }
54  else if (max == r && g < b)
55  {
56  *h = 60 * (g - b) / (max - min) + 360;
57  }
58  else if (max == g)
59  {

61  }
62  else if (max == b)
63  {
64  *h = 60 * (r - g) / (max - min) + 240;
65  }
66 
67

161 AWPRESULT _awp_hsv_to_rgb(AWPDOUBLE h, AWPDOUBLE s, AWPDOUBLE v, AWPBYTE* rr, AWPBYTE* gg, AWPBYTE* bb)
162 {
163  AWPINT i;
164  AWPDOUBLE f, p, q, t;
165  AWPDOUBLE r, g, b;
166 
167  if (s == 0)
168  {
169  // achromatic (grey)
170  *rr = *gg = *bb = (AWPBYTE)(255 * v);
171  return AWP_OK;
172  }
173  h *= 6; // sector 0 to 5
174  i = (AWPINT)floor(h);
175  f = h - i;
176  p = v * (1 - s);
177  q = v * (1 - s * f);
178  t = v * (1 - s * (1 - f));
179  switch (i) {
180  case 0:
181  r = v;
182  g = t;
183  b = p;
184  break;
185  case 1:
186  r = q;
187  g = v;
188  b = p;
189  break;
190  case 2:
191  r = p;
192  g = v;
193  b = t;
194  break;
195  case 3:
196  r = p;
197  g = q;
198  b = v;
199  break;
200  case 4:
201  r = t;
202  g = p;
203  b = v;
204  break;
205  default: // case 5:
206  r = v;
207  g = p;
208  b = q;
209  break;
210  }
211 
212  *rr = (AWPBYTE)(255 * r);
213  *gg = (AWPBYTE)(255 * g);
214  *bb = (AWPBYTE)(255 * b);
215 
216  return AWP_OK;
217 }
218 AWPRESULT _awp_hsl_to_rgb(AWPDOUBLE h, AWPDOUBLE s, AWPDOUBLE l, AWPBYTE* rr, AWPBYTE* gg, AWPBYTE* bb)
219 {
220  AWPDOUBLE v;
221  AWPDOUBLE r, g, b;
222 
223  r = g = b = 0.0;
224 
225  v = (l <= 0.5) ? (l * (1.0 + s)) : (l + s - l * s);
226 
227  if (v <= 0) {
228  r = g = b = 0.0;
229  }
230  else {
231  AWPDOUBLE m;
232  AWPDOUBLE sv;
233  AWPINT sextant;
234  AWPDOUBLE fract, vsf, mid1, mid2;
235 
236  m = l + l - v;
237  sv = (v - m) / v;
238  h *= 6.0;
239  sextant = (AWPINT)h;
240  fract = h - sextant;
241  vsf = v * sv * fract;
242  mid1 = m + vsf;
243  mid2 = v - vsf;
244  switch (sextant) {
245  case 0: r = v; g = mid1; b = m; break;
246  case 1: r = mid2; g = v; b = m; break;
247  case 2: r = m; g = v; b = mid1; break;
248  case 3: r = m; g = mid2; b = v; break;
249  case 4: r = mid1; g = m; b = v; break;
250  case 5: r = v; g = m; b = mid2; break;
251  }
252  }
253 
254  *rr = (AWPBYTE)(255 * r);
255  *gg = (AWPBYTE)(255 * g);
256  *bb = (AWPBYTE)(255 * b);
257 
258  return AWP_OK;
259 }
