112 AWPRESULT awpJpegToImage( const AWPBYTE* pJpegData, AWPDWORD length, awpImage** ppImage )
113 {
114  AWPRESULT res;
115 
116  struct my_error_mgr jpeg_err;
117  struct jpeg_decompress_struct in_jpeg;
118  AWPBYTE* pDst;
119  JSAMPARRAY buffer; /* Output row buffer */
120  AWPINT row_stride;
121  AWPINT rows;
122  JDIMENSION i;
123  AWPBYTE* src = NULL;
124 
125  //set error function
126  in_jpeg.err = jpeg_std_error( &jpeg_err.pub );
127  jpeg_err.pub.error_exit = my_error_exit;
128 
129  if (setjmp(jpeg_err.setjmp_buffer))
130  {
131  jpeg_destroy_decompress(&in_jpeg);
132  return AWP_BADIMAGE_FORMAT;
133  }
134 
135  res = AWP_OK;
136 
137  jpeg_create_decompress( &in_jpeg );
138 
139  /* read from buffer */
140  jpeg_buffer_src( &in_jpeg, (char*)pJpegData, length );
141  jpeg_read_header( &in_jpeg, TRUE );
142 
143 
144  if(!(in_jpeg.out_color_space != JCS_RGB || in_jpeg.out_color_space != JCS_GRAYSCALE)){
145  res = AWP_NOTSUPPORT;
146  _ERR_EXIT_
147  }
148 
149  jpeg_start_decompress( &in_jpeg );
150 
151  row_stride = in_jpeg.output_width*in_jpeg.output_components;
152  rows = in_jpeg.output_height;
153 
154  /* create awp image */
155  _CHECK_RESULT_((res = awpCreateImage(ppImage,(AWPWORD)in_jpeg.output_width,(AWPWORD)in_jpeg.output_height,(AWPBYTE)(in_jpeg.out_color_space==JCS_RGB?3:1),AWP_BYTE)))
156 
157  buffer = (*in_jpeg.mem->alloc_sarray)
158  ((j_common_ptr) &in_jpeg, JPOOL_PERMANENT, row_stride,rows );
159 
160 
161  pDst = (AWPBYTE*)((*ppImage)->pPixels);
162  while (in_jpeg.output_scanline < in_jpeg.output_height)
163  {
164  jpeg_read_scanlines(&in_jpeg, buffer, 1);
165  src = (AWPBYTE*)buffer[0];
166  for (i = 0; i < in_jpeg.output_width; i++)
167  {
168  pDst[3*i + 2] = src[0];
169  pDst[3*i + 1] = src[1];
170  pDst[3*i] = src[2];
171  src += 3;
172  }
173 
174  pDst += row_stride;
175  }
176 
177 
178 CLEANUP:
179 
180  jpeg_finish_decompress( &in_jpeg );
181  jpeg_destroy_decompress( &in_jpeg );
182 
183  return res;
184
