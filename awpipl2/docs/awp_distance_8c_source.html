165 /*
166 Calculates the distance between two images. The result is written to the variable res.
167 Images should have one data channel of the same size and
168 be of the same type.
169 */
170 AWPRESULT awpDistance(awpImage* src1, awpImage* src2, AWPBYTE options, AWPDOUBLE* res)
171 {
172  AWPRESULT result = AWP_OK;
173 
174  if (src1 == NULL || src2 == NULL || res == NULL)
175  return AWP_BADARG;
176  if (src1->sSizeX != src2->sSizeX)
177  return AWP_BADARG;
178  if (src1->sSizeY != src2->sSizeY)
179  return AWP_BADARG;
180  if (src1->dwType != src2->dwType)
181  return AWP_BADARG;
182  if (src1->bChannels != 1)
183  return AWP_BADARG;
184  if (src1->bChannels != src2->bChannels)
185  return AWP_BADARG;
186 
187  *res = 0;
188 
189  switch(options)
190  {
191  case AWP_DIST_L1:
192  result = awpL1Distance(src1, src2, res);
193  break;
194 
195  case AWP_DIST_L2:
196  result = awpL2Distance(src1, src2, res);
197  break;
198 
199  case AWP_DIST_NCC:
200  result = awpNCCDistance(src1, src2, res);
201  break;
202  case AWP_DIST_X2:
203  result = awpX2Distance(src1, src2, res);
204  break;
205  case AWP_DIST_MSE:
206  result = awpL2Distance(src1, src2, res);
207  _CHECK_RESULT_(result)
208  *res = *res / (AWPDOUBLE)(src1->sSizeX*src1->sSizeY);
209  break;
210 
211  case AWP_DIST_RMSE:
212  result = awpL2Distance(src1, src2, res);
213  _CHECK_RESULT_(result)
214  *res = *res / (AWPDOUBLE)(src1->sSizeX*src1->sSizeY);
215  *res = sqrt(*res);
216  break;
217  default:
218  result = AWP_BADARG;
219  }
220 
221 CLEANUP:
222  return result;
223 }
