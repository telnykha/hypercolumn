22 AWPRESULT awpCheckImage(const awpImage* pImage)
23 {
24  /*check pointer*/
25  if (pImage == NULL)
26  return AWP_BADMEMORY;
27  /*check magic number*/
28  if (pImage->nMagic != AWP_MAGIC2)
29  return AWP_NOTAWPIMAGE;
30  /*check channles*/
31  if (pImage->bChannels == 0)
32  return AWP_BADIMAGE_FORMAT;
33  /*check sizes*/
34  if (pImage->sSizeX == 0 ||pImage->sSizeY == 0)
35  return AWP_BADSIZES;
36  /*check format*/
37  if (pImage->dwType != AWP_BYTE &&
38  pImage->dwType != AWP_SHORT &&
39  pImage->dwType != AWP_FLOAT &&
40  pImage->dwType != AWP_DOUBLE )
41  return AWP_BADIMAGE_FORMAT;
42  /*check pixels*/
43  if (pImage->pPixels == NULL)
44  return AWP_BADMEMORY;
45 
46  return AWP_OK;
47

66 AWPRESULT awpCreateImage(awpImage** ppImage,AWPWORD width,AWPWORD height,AWPBYTE channels, AWPDWORD dwType)
67 {
68  awpImage* pImage;
69  AWPINT size;
70  if( ppImage == NULL )
71  return AWP_BADARG;
72  if (channels == 0 || width == 0 || height == 0)
73  return AWP_BADARG;
74  *ppImage = NULL;
75  /*todo: this code is subject to change
76  dwType may be not only just a set of channels....*/
77  switch(dwType)
78  {
79  case AWP_BYTE: size = channels*sizeof(AWPBYTE)*width*height;break;
80  case AWP_SHORT:size = channels*sizeof(AWPSHORT)*width*height;break;
81  case AWP_FLOAT:size = channels*sizeof(AWPFLOAT)*width*height;break;
82  case AWP_DOUBLE:size = channels*sizeof(AWPDOUBLE)*width*height;break;
83  default:
84  return AWP_BADARG;
85  }
86  /*alloc memory for header*/
87  pImage = NULL;
88  pImage = (awpImage*)malloc(sizeof(awpImage));
89  if (pImage == NULL)
90  return AWP_BADMEMORY;
91 
92  pImage->nMagic = AWP_MAGIC2;
93  pImage->bChannels = channels;
94  pImage->sSizeX = width;
95  pImage->sSizeY = height;
96  pImage->dwType = dwType;
97  pImage->pPixels = NULL;
98 
99  pImage->pPixels = malloc(size);
100  if (pImage->pPixels == NULL)
101  {
102  free(pImage);
103  return AWP_BADMEMORY;
104  }
105  /*fill memory*/
106  memset((AWPBYTE*)pImage->pPixels,0,size);
107  *ppImage = pImage;
108  return AWP_OK;
109 }

132 
133 
134  bpp = channels > 3 ? 3 : channels;
135  for (i = 0; i < width*height*channels; i+= channels)
136  {
137  sum = 0;

139  sum += data[i + k];
140  sum /= bpp;

143  }
144
