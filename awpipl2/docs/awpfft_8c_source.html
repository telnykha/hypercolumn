22 AWPRESULT awpFFT(awpImage** pImage,awpImage** pImageRe,awpImage** pImageIm,AWPINT iFftDir)

24  AWPRESULT res;

31 
32  res = AWP_OK;/*initialize result */
33  in=NULL;out=NULL;
34  p=NULL;
35 
36  switch(iFftDir){
37  case AWP_FFT_FORWARD:
38  /* check image */
39  _CHECK_RESULT_((res = awpCheckImage(*pImage)))
40 
41  /* only AWPDOUBLE image support */
42  if((*pImage)->dwType!=AWP_DOUBLE){
43  res = AWP_NOTSUPPORT;
44  _ERR_EXIT_
45  }
46 
47  *pImageRe=NULL;
48  *pImageIm=NULL;
49 
50  W=(*pImage)->sSizeX;
51  H=(*pImage)->sSizeY;
52 
53  break;
54  case AWP_FFT_INVERSE:
55 
56  /* check image */
57  _CHECK_RESULT_((res = awpCheckImage(*pImageRe)))
58  _CHECK_RESULT_((res = awpCheckImage(*pImageIm)))
59 
60  /* only AWPDOUBLE image support */
61  if((*pImageRe)->dwType != AWP_DOUBLE || (*pImageIm)->dwType != AWP_DOUBLE ){
62  res = AWP_NOTSUPPORT;
63  _ERR_EXIT_
64  }
65 
66  if((*pImageRe)->sSizeX != (*pImageIm)->sSizeX || (*pImageRe)->sSizeY != (*pImageIm)->sSizeY ){
67  res = AWP_BADARG;
68  _ERR_EXIT_
69  }
70 
71  W=(*pImageRe)->sSizeX;
72  H=(*pImageRe)->sSizeY;
73 
74  *pImage=NULL;
75 
76  break;
77  default:
78  res = AWP_BADARG;
79  _ERR_EXIT_
80  }
81 
82  /* allocate memory for fft */
83  out =(fftw_complex*)fftw_malloc(sizeof(fftw_complex)*W*(H/2+1));
84 
85  switch(iFftDir){
86  case AWP_FFT_FORWARD:
87  in=(AWPDOUBLE*)(*pImage)->pPixels;
88 
89  /* create plan */
90  p= fftw_plan_dft_r2c_2d(W,H,in,out,FFTW_ESTIMATE);
91 
92  break;
93  case AWP_FFT_INVERSE:
94  /* fill dat from image */
95  pRe=(AWPDOUBLE*)(*pImageRe)->pPixels;
96  pIm=(AWPDOUBLE*)(*pImageIm)->pPixels;
97 
98  for(i=0;i<H;i++){
99  for(j=0;j<W;j++){
100  out[i*W+j][0]=pRe[i*W+j];
101  out[i*W+j][1]=pIm[i*W+j];
102  }
103  }
104 
105  /* create image */
106  _CHECK_RESULT_((res=awpCreateImage(pImage,W,H,1,AWP_DOUBLE)))
107 
108  in=(AWPDOUBLE*)(*pImage)->pPixels;
109 
110  /* create plan */
111  p= fftw_plan_dft_c2r_2d(W,H,out,in,FFTW_ESTIMATE);
112 
113  break;
114  }
115 
116  if(p==NULL){
117  res = AWP_BADMEMORY;
118  _ERR_EXIT_
119  }
120 
121  /* execute fft transofm */
122  fftw_execute(p);
123 
124  if(iFftDir==AWP_FFT_FORWARD){
125  /* create image */
126  _CHECK_RESULT_((res=awpCreateImage(pImageRe,W,H/2+1,1,AWP_DOUBLE)))
127  _CHECK_RESULT_((res=awpCreateImage(pImageIm,W,H/2+1,1,AWP_DOUBLE)))
128 
129  pRe=(AWPDOUBLE*)(*pImageRe)->pPixels;
130  pIm=(AWPDOUBLE*)(*pImageIm)->pPixels;
131 
132  for(i=0;i<W;i++){
133  count=W*(H/2+1);
134  for(j=0;j<H/2+1;j++){
135  pRe[j*W+i]=out[count+j][0];
136  pIm[j*W+i]=out[count+j][1];
137  }
138  }
139  }
140 
141 CLEANUP:
142  if(p!=NULL){
143  fftw_destroy_plan(p);
144  }
145 
146  return res;
147 }

174  res = AWP_NOTSUPPORT;
175  _ERR_EXIT_

181

183  cmx=pMask->sSizeX/2;
184  cmy=pMask->sSizeY/2;
185 
186  m=(AWPDOUBLE*)pBigMask->pPixels;
187  ms=(AWPDOUBLE*)pMask->pPixels;
188 
189  /* fill big mask */
190  /* fill top left angle big mask */
191  for(i=0;i<cmy+1;i++){
192  for(j=0;j<cmx+1;j++){
193  m[i*pBigMask->sSizeX+j]=ms[(i+cmy-1)*pMask->sSizeX+j+cmx-1];
194  }
195  }
196 
197  /* fill top right angle big mask */
198  for(i=0;i<cmy+1;i++){
199  for(j=0;j<cmx;j++){
200  m[i*pBigMask->sSizeX+(pBigMask->sSizeX-cmx+j)]=ms[(i+cmy)*pMask->sSizeX+j];
201  }
202  }
203 
204  for(i=0;i<cmy;i++){
205  for(j=0;j<cmx+1;j++){
206  /* fill bottom left angle big mask */
207  m[(pBigMask->sSizeY-cmy+i)*pBigMask->sSizeX+j]=ms[i*pMask->sSizeX+j+cmx];
208  }
209  }
210 
211  for(i=0;i<cmy;i++){
212  for(j=0;j<cmx;j++){
213  /* fill bottom right angle big mask */
214  m[(pBigMask->sSizeY-cmy+i)*pBigMask->sSizeX+(pBigMask->sSizeX-cmx+j)]=ms[i*pMask->sSizeX+j];
215  }
216  }
217 
218  awpCopyImage(pBigMask,&pImCp);
219  awpConvert(pImCp,AWP_CONVERT_TO_BYTE_WITH_NORM);
220  awpSaveImage("mask_big.jpg",pImCp);
221  awpReleaseImage(&pImCp);
222 
223  im=(AWPDOUBLE*)pSrcImage->pPixels;
224 
225  W=pSrcImage->sSizeX;
226  H=pSrcImage->sSizeY;
227  scale = 1.0/(W*H);
228 
229  /* allocate data for complex data */
230  IM = (fftw_complex*)fftw_malloc(sizeof(fftw_complex)*H*(W/2+1));
231  M = (fftw_complex*)fftw_malloc(sizeof(fftw_complex)*H*(W/2+1));
232  C = (fftw_complex*)fftw_malloc(sizeof(fftw_complex)*H*(W/2+1));
233 
234  /* create plan for image */
235  pi = fftw_plan_dft_r2c_2d(H,W,im,IM,FFTW_ESTIMATE);
236 
237  if(pi==NULL){
238  res = AWP_BADMEMORY;
239  _ERR_EXIT_
240  }
241 
242  /* execute fft transofm */
243  fftw_execute(pi);
244 
245  /* create plan for image */
246  pm = fftw_plan_dft_r2c_2d(H,W,m,M,FFTW_ESTIMATE);
247 
248  if(pm==NULL){
249  res = AWP_BADMEMORY;
250  _ERR_EXIT_
251  }
252 
253  /* execute fft transofm */
254  fftw_execute(pm);
255 
256  for(i=0;i<H;i++){
257  ij_j=i*(W/2+1);
258  for(j=0;j<W/2+1;j++){
259  ij=ij_j+j;
260  C[ij][0]=( IM[ij][0]*M[ij][0] - IM[ij][1]*M[ij][1] )*scale;
261  C[ij][1]=( IM[ij][0]*M[ij][1] + IM[ij][1]*M[ij][0] )*scale;
262  }
263  }
264 
265  _CHECK_RESULT_((res=awpCreateImage(pDstImage,pSrcImage->sSizeX,pSrcImage->sSizeY,1,AWP_DOUBLE)))
266 
267  ic=(AWPDOUBLE*)(*pDstImage)->pPixels;
268 
269  pc = fftw_plan_dft_c2r_2d(H,W,C,ic,FFTW_ESTIMATE);
270 
271  if(pc==NULL){
272  res = AWP_BADMEMORY;
273  _ERR_EXIT_
274  }
275 
276  /* execute fft transofm */
277  fftw_execute(pc);
278 
279 CLEANUP:
280

282  fftw_destroy_plan(pi);
283  }
