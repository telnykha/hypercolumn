21 void _awpPolynomCalculate(AWPINT K, AWPINT N, AWPDOUBLE*x, AWPDOUBLE*y, AWPDOUBLE*a)
22 {
23  AWPINT i, j, k;
24  AWPDOUBLE s, M;
25  AWPDOUBLE *b = NULL;
26  AWPDOUBLE** sums = NULL;
27  memset(a, 0, (K+1)*sizeof(AWPDOUBLE));
28  b = (double*)malloc(N * sizeof(AWPDOUBLE));
29  sums = (AWPDOUBLE**)malloc(N* sizeof(AWPDOUBLE*));//creating a two-dimensional array
30  for (i = 0; i < N; i++)
31  {
32  sums[i] = (AWPDOUBLE*)malloc(N* sizeof(AWPDOUBLE));
33  }
34  //fill the coefficients of the system of equations
35  for (i = 0; i<K + 1; i++)
36  {
37  for (j = 0; j<K + 1; j++)
38  {
39  sums[i][j] = 0;
40  for (k = 0; k<N; k++)
41  sums[i][j] += Power(x[k], i + j);
42  }
43  }
44  //fill in the free members column
45  for (i = 0; i<K + 1; i++)
46  {
47  b[i] = 0;
48  for (k = 0; k<N; k++)
49  b[i] += Power(x[k], i)*y[k];
50  }
51  //we use the Gauss method to bring the system matrix to a triangular shape
52  for (k = 0; k<K + 1; k++)
53  {
54  for (i = k + 1; i<K + 1; i++)
55  {
56  M = sums[i][k] / sums[k][k];
57  for (j = k; j<K + 1; j++)
58  sums[i][j] -= M * sums[k][j];
59  b[i] -= M*b[k];
60  }
61  }
62  //calculate the coefficients of the approximating polynomial
63  for (i = K; i >= 0; i--)
64  {
65  s = 0;
66  for (j = i; j<K + 1; j++)
67  s += sums[i][j] * a[j];
68  a[i] = (b[i] - s) / sums[i][i];
69  }
70  free(b);
71  for (i = 0; i < N; i++)
72  {
73  free(sums[i]);
74  }
75  free(sums);
76 }
