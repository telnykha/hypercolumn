402 __inline static double _awpAtan(double dx, double dy)
403 {
404  AWPDOUBLE t = 0;
405  if (dx == 0)
406  return 0;
407  t = fabs(dy / dx);
408  if (dx > 0 && dy >= 0)
409  return atan(t);
410  else if (dx >=0 && dy < 0)
411  return 1.5*AWP_PI + atan(t);
412  else if (dx < 0 && dy < 0)
413  return AWP_PI + atan(t);
414  else
415  return 0.5*AWP_PI + atan(t);
